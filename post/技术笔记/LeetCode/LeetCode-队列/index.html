        //å¤åŸåŒç«¯é˜Ÿåˆ—
        while (!stack.isEmpty()) {
            int temp = (int) stack.pop();
            deque.offerLast(temp);
        }

        for(int i = 0; k + i <= length; i++) {

            //è·å–åœ¨å½“å‰çª—å£çš„æœ€å¤§å€¼
            boolean flag = false; //æ˜¯å¦è·å–åˆ°å½“å‰çª—å£æœ€å¤§å€¼çš„æ ‡å¿—
            while(!flag) {
                if(i <= priorityQueue.peek()[1] && priorityQueue.peek()[1]  < k + i) {
                    res[i] = priorityQueue.peek()[0];
                    flag = true;
                } else {
                    priorityQueue.poll();
                }
            }
            //ç§»åŠ¨çª—å£
            if(k + i < length) {
                deque.pollFirst();
                deque.offerLast(nums[k + i]);
            }
            //å°†æœ€æ–°åŠ å…¥çª—å£çš„å…ƒç´ åŠ å…¥ä¼˜å…ˆé˜Ÿåˆ—
            int temp = (int) deque.getLast();
            priorityQueue.offer(new int[]{temp, k + i});
        }

        return res;
    }
